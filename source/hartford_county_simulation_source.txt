function mulberry32(s){return function(){s|=0;s=s+0x6D2B79F5|0;let t=Math.imul(s^s>>>15,1|s);t=t+Math.imul(t^t>>>7,61|t)^t;return((t^t>>>14)>>>0)/4294967296;};}
function dist(a,b){const R=3958.8,toR=d=>d*Math.PI/180;const dLa=toR(b.lat-a.lat),dLo=toR(b.lon-a.lon);const s=Math.sin(dLa/2)**2+Math.cos(toR(a.lat))*Math.cos(toR(b.lat))*Math.sin(dLo/2)**2;return 2*R*Math.asin(Math.sqrt(s));}
function pointInRing(lon,lat,ring){let inside=false;for (let i=0,j=ring.length-1;i<ring.length;j=i++){const xi=ring[i][0],yi=ring[i][1],xj=ring[j][0],yj=ring[j][1];if (((yi>lat)!==(yj>lat)) && (lon<(xj-xi)*(lat-yi)/(yj-yi)+xi)) inside=!inside;}return inside;}
function pointInGeo(lon,lat,geo){const polys=geo.type==="Polygon"?[geo.coordinates]:geo.coordinates;for (const poly of polys){if (!pointInRing(lon,lat,poly[0])) continue;let h=false;for (let i=1;i<poly.length;i++) if (pointInRing(lon,lat,poly[i])){h=true;break;}if (!h) return true;}return false;}
function geoBounds(geo){let minLa=90,maxLa=-90,minLo=180,maxLo=-180;const v=r=>r.forEach(([lo,la])=>{minLa=Math.min(minLa,la);maxLa=Math.max(maxLa,la);minLo=Math.min(minLo,lo);maxLo=Math.max(maxLo,lo);});if (geo.type==="Polygon") v(geo.coordinates[0]); else geo.coordinates.forEach(p=>v(p[0]));return L.latLngBounds([minLa,minLo],[maxLa,maxLo]);}

// --- Spatial accelerators (built once after boundary loads) ---
// 1) inside-bitmap: a coarse grid over the county bbox where each cell stores
//    a precomputed point-in-polygon result. Reduces O(P) ring tests to O(1) lookups.
// 2) sqEuclid: cheap k-means distance (no trig).
const GRID_N=256;
let _insideBits=null, _gLatMin=0,_gLatMax=0,_gLonMin=0,_gLonMax=0,_gDLat=0,_gDLon=0;
let _bitmapReady=false;
function buildInsideBitmap(geo){
  try {
    const b=geoBounds(geo);
    _gLatMin=b.getSouth(); _gLatMax=b.getNorth();
    _gLonMin=b.getWest();  _gLonMax=b.getEast();
    _gDLat=(_gLatMax-_gLatMin)/GRID_N;
    _gDLon=(_gLonMax-_gLonMin)/GRID_N;
    _insideBits=new Uint8Array(GRID_N*GRID_N);
    let count=0;
    for (let i=0;i<GRID_N;i++){
      const lat=_gLatMin+(i+0.5)*_gDLat;
      for (let j=0;j<GRID_N;j++){
        const lon=_gLonMin+(j+0.5)*_gDLon;
        if (pointInGeo(lon,lat,geo)){ _insideBits[i*GRID_N+j]=1; count++; }
      }
    }
    _bitmapReady = count > 0;
    console.log(`inside bitmap: ${count}/${GRID_N*GRID_N} cells inside county`);
  } catch(e){
    console.warn("bitmap build failed, will fall back to pointInGeo:", e);
    _bitmapReady=false;
  }
}
function insideFast(lon,lat){
  // Fall back to pointInGeo if bitmap isn't ready — keeps generation correct even on bitmap failure.
  if (!_bitmapReady) return countyGeo ? pointInGeo(lon,lat,countyGeo) : true;
  if (lat<_gLatMin||lat>=_gLatMax||lon<_gLonMin||lon>=_gLonMax) return false;
  const i=(lat-_gLatMin)/_gDLat|0, j=(lon-_gLonMin)/_gDLon|0;
  return _insideBits[i*GRID_N+j]===1;
}
function sqEuclid(a,b){const dx=a.lat-b.lat,dy=a.lon-b.lon;return dx*dx+dy*dy;}

// --- Step 1: Substation placement via weighted k-means on demand points ---
function buildDemandPoints(rnd){
  const pts=[];
  // (a) Population-weighted points clustered around town centers.
  TOWNS.forEach(t=>{
    const n=Math.max(8,Math.round(Math.sqrt(t.pop)/2));
    const w=t.pop/n;
    for (let i=0;i<n;i++){
      const r=0.005+rnd()*0.015, a=rnd()*Math.PI*2;
      pts.push({lat:t.lat+Math.sin(a)*r*0.85,lon:t.lon+Math.cos(a)*r,w});
    }
  });
  // (b) Uniform-area points across the entire county polygon.
  // This pulls k-means centroids into rural areas so substations reach every corner.
  if (_bitmapReady){
    const uniformN = 1200;
    const uniformWeightTotal = TOTAL_POP * 0.35;   // 35% of total weight goes to area
    const wPerPt = uniformWeightTotal / uniformN;
    let placed=0, tries=0;
    while (placed < uniformN && tries < uniformN*10){
      tries++;
      const lat = _gLatMin + rnd()*(_gLatMax-_gLatMin);
      const lon = _gLonMin + rnd()*(_gLonMax-_gLonMin);
      if (!insideFast(lon,lat)) continue;
      pts.push({lat,lon,w:wPerPt});
      placed++;
    }
  }
  return pts;
}
function kmeans(pts,k,geo,rnd,iters=15){
  // Flat typed arrays for hot loops
  const N=pts.length;
  const pLat=new Float64Array(N), pLon=new Float64Array(N), pW=new Float64Array(N);
  for (let i=0;i<N;i++){pLat[i]=pts[i].lat;pLon[i]=pts[i].lon;pW[i]=pts[i].w;}
  const cLat=new Float64Array(k), cLon=new Float64Array(k);
  // k-means++ seeding with squared-Euclidean (no trig)
  const first=Math.floor(rnd()*N);
  cLat[0]=pLat[first]; cLon[0]=pLon[first];
  const minD=new Float64Array(N); for (let i=0;i<N;i++){const dx=pLat[i]-cLat[0],dy=pLon[i]-cLon[0];minD[i]=dx*dx+dy*dy;}
  for (let c=1;c<k;c++){
    let sum=0; for (let i=0;i<N;i++) sum+=minD[i]*pW[i];
    let r=rnd()*sum, idx=N-1;
    for (let i=0;i<N;i++){r-=minD[i]*pW[i];if (r<=0){idx=i;break;}}
    cLat[c]=pLat[idx]; cLon[c]=pLon[idx];
    for (let i=0;i<N;i++){const dx=pLat[i]-cLat[c],dy=pLon[i]-cLon[c];const d=dx*dx+dy*dy;if (d<minD[i]) minD[i]=d;}
  }
  // Lloyd iterations
  const sL=new Float64Array(k), sN=new Float64Array(k), sW=new Float64Array(k);
  for (let it=0;it<iters;it++){
    sL.fill(0); sN.fill(0); sW.fill(0);
    for (let i=0;i<N;i++){
      const lat=pLat[i],lon=pLon[i],w=pW[i];
      let b=0,bd=Infinity;
      for (let c=0;c<k;c++){const dx=lat-cLat[c],dy=lon-cLon[c];const d=dx*dx+dy*dy;if (d<bd){bd=d;b=c;}}
      sL[b]+=lat*w; sN[b]+=lon*w; sW[b]+=w;
    }
    for (let c=0;c<k;c++){
      if (sW[c]<=0) continue;
      let nl=sL[c]/sW[c], ln=sN[c]/sW[c];
      // Snap centroid back inside via the bitmap (was the slow path)
      if (geo && !insideFast(ln,nl)){
        let bestI=0,bd=Infinity;
        for (let i=0;i<N;i++){const dx=pLat[i]-nl,dy=pLon[i]-ln;const d=dx*dx+dy*dy;if (d<bd){bd=d;bestI=i;}}
        nl=pLat[bestI]; ln=pLon[bestI];
      }
      cLat[c]=nl; cLon[c]=ln;
    }
  }
  const out=new Array(k); for (let c=0;c<k;c++) out[c]={lat:cLat[c],lon:cLon[c]};
  return out;
}

// --- Step 2: Generate feeders (radial backbones) + laterals (random walks) ---
function generateGrid(){
  const seed=+seedI.value|0;
  const rnd=mulberry32(seed+1);
  const k=+kS.value, feedersPerSub=+fS.value;
  status.textContent=`Placing ${k} substations…`;

  const demand=buildDemandPoints(rnd);
  const centers=kmeans(demand,k,countyGeo,rnd);

  substations=centers.map((c,i)=>({lat:c.lat,lon:c.lon,color:PALETTE[i%PALETTE.length],popServed:0}));

  // Assign each demand point to nearest substation → estimate popServed
  demand.forEach(d=>{
    let b=0,bd=Infinity;
    substations.forEach((s,i)=>{const dd=dist(d,s);if (dd<bd){bd=dd;b=i;}});
    substations[b].popServed+=d.w;
  });

  status.textContent=`Generating feeders and laterals…`;
  feeders=[]; laterals=[];

  // For each substation: f feeder backbones radiating outward
  substations.forEach((s,si)=>{
    const popPerFeeder=s.popServed/feedersPerSub;
    for (let f=0;f<feedersPerSub;f++){
      const baseAng=(f/feedersPerSub)*Math.PI*2+rnd()*0.4;
      let lat=s.lat,lon=s.lon,ang=baseAng;
      const pts=[[lat,lon]];
      const segs=8+Math.floor(rnd()*8);
      for (let g=0;g<segs;g++){
        ang+=(rnd()-0.5)*0.5;
        const step=0.004+rnd()*0.005;
        lon+=Math.cos(ang)*step; lat+=Math.sin(ang)*step;
        if (!insideFast(lon,lat)) break;
        pts.push([lat,lon]);
      }
      if (pts.length<2) continue;
      const fIdx=feeders.length;
      feeders.push({subIdx:si,pts,popServed:popPerFeeder,color:s.color});

      // Laterals: branch from each midpoint of the feeder
      const lateralsPerFeeder=4+Math.floor(rnd()*5);
      for (let l=0;l<lateralsPerFeeder;l++){
        const anchor=pts[1+Math.floor(rnd()*(pts.length-1))];
        let lLat=anchor[0],lLon=anchor[1],lAng=rnd()*Math.PI*2;
        const lPts=[[lLat,lLon]];
        const lSegs=3+Math.floor(rnd()*5);
        for (let g=0;g<lSegs;g++){
          lAng+=(rnd()-0.5)*1.0;
          const lStep=0.0015+rnd()*0.0025;
          lLon+=Math.cos(lAng)*lStep; lLat+=Math.sin(lAng)*lStep;
          if (!insideFast(lLon,lLat)) break;
          lPts.push([lLat,lLon]);
        }
        if (lPts.length<2) continue;
        laterals.push({feederIdx:fIdx,pts:lPts,popServed:popPerFeeder/lateralsPerFeeder});
      }
    }
  });

  renderGrid();
  console.log(`Generated: subs=${substations.length}, feeders=${feeders.length}, laterals=${laterals.length}`);
  if (feeders.length === 0) {
    status.textContent = `WARNING: 0 feeders generated. Check console for errors. Substations: ${substations.length}.`;
    return;
  }
  status.textContent=`Grid: ${substations.length} substations, ${feeders.length} feeders, ${laterals.length} laterals. Ready to simulate storm.`;
  // Reset downstream state
  storm=null; plan=null;
  outageLayer.clearLayers(); repairLayer.clearLayers(); depotLayer.clearLayers();
  affectedEl.textContent='0'; affectedPctEl.textContent='0% of county population';
  outageCountEl.textContent='0'; restoreTimeEl.textContent='—'; restoreDetailEl.textContent='— customers restored';
}

function renderGrid(){
  lateralLayer.clearLayers(); feederLayer.clearLayers(); subLayer.clearLayers();
  laterals.forEach(l=>L.polyline(l.pts,{renderer:gridCanvas,color:"#bfc4cb",weight:0.6,opacity:0.75,interactive:false}).addTo(lateralLayer));
  feeders.forEach(f=>L.polyline(f.pts,{renderer:gridCanvas,color:f.color,weight:1.3,opacity:0.95,interactive:false}).addTo(feederLayer));
  substations.forEach((s,i)=>{
    const icon=L.divIcon({className:"sub-star",html:starSVG(s.color,14),iconSize:[14,14],iconAnchor:[7,7]});
    L.marker([s.lat,s.lon],{icon,interactive:true}).addTo(subLayer)
      .bindTooltip(`Substation ${i+1} — ${Math.round(s.popServed).toLocaleString()} customers`);
  });
  // Layer order after re-render: mask hides outside-county; boundary lives on its own top pane.
  if (window._mask) window._mask.bringToFront();
}

// --- Step 3: Storm — pick N outage points on the network ---
function simulateStorm(){
  if (!feeders.length){status.textContent="Generate the grid first.";return;}
  const seed=+seedI.value|0;
  const rnd=mulberry32(seed*7919+13);
  const N=+oS.value;
  const realistic=document.getElementById('realisticMode').checked;
  // Sectionalizer model: in reality, protective devices isolate just the broken
  // segment between two switches rather than killing the whole downstream branch.
  // Real impact per outage is ~half the simple "drop everything downstream" model.
  const popMultiplier = realistic ? 0.5 : 1.0;
  // Tier-1 critical-facility share: hospitals, water plants, fire stations, etc.
  // ~2% of outage locations are flagged for top-priority restoration.
  const critRate = realistic ? 0.02 : 0;

  const outages=[];
  // Outages are placed on a random line segment (feeder or lateral, weighted by laterals for realism)
  const allSegs=[];
  feeders.forEach((f,fi)=>{for (let s=0;s<f.pts.length-1;s++) allSegs.push({kind:'f',fi,s,popServed:f.popServed/(f.pts.length-1)});});
  laterals.forEach((l,li)=>{for (let s=0;s<l.pts.length-1;s++) allSegs.push({kind:'l',li,s,popServed:l.popServed/(l.pts.length-1)});});

  for (let i=0;i<N;i++){
    const seg=allSegs[Math.floor(rnd()*allSegs.length)];
    const t=rnd();
    const arr=seg.kind==='f'?feeders[seg.fi].pts:laterals[seg.li].pts;
    const a=arr[seg.s], b=arr[seg.s+1];
    const lat=a[0]+(b[0]-a[0])*t, lon=a[1]+(b[1]-a[1])*t;
    outages.push({
      lat, lon, kind:seg.kind, fi:seg.fi, li:seg.li,
      popLoss: seg.popServed * popMultiplier,
      critical: rnd() < critRate,
    });
  }

  // Customers affected = sum of popLoss, capped at county population.
  let totalCust=outages.reduce((s,o)=>s+o.popLoss,0);
  totalCust=Math.min(totalCust,TOTAL_POP);

  storm={outages,totalCust};
  plan=null;

  // Render
  outageLayer.clearLayers(); repairLayer.clearLayers(); depotLayer.clearLayers();
  outages.forEach(o=>{
    // Critical-facility outages (tier 1) get a slightly bigger, brighter marker
    // so they stand out from the dark-red cloud of ordinary failures.
    L.circleMarker([o.lat,o.lon],{
      renderer:outageCanvas,
      radius: o.critical ? 4 : 2.5,
      color: o.critical ? "#facc15" : "#000",
      weight: o.critical ? 1.2 : 0.5,
      fillColor: o.critical ? "#dc2626" : "#7f1d1d",
      fillOpacity: 0.92,
      interactive:false
    }).addTo(outageLayer);
  });

  affectedEl.textContent=Math.round(totalCust).toLocaleString();
  affectedPctEl.textContent=(100*totalCust/TOTAL_POP).toFixed(1)+"% of county population";
  outageCountEl.textContent=N.toLocaleString();
  restoreTimeEl.textContent='—'; restoreDetailEl.textContent='— customers restored';
  status.textContent=`Storm: ${N.toLocaleString()} outages, ~${Math.round(totalCust).toLocaleString()} customers without power.`;
}

// --- Spatial grid hash for nearest-outage queries ---
// At 25K outages × 5K crews, the per-crew linear scan dominates. This grid
// indexes outages into ~500m cells; nearest queries inspect just a handful of
// cells outward until the next ring can't possibly contain a closer point.
// Average query cost is roughly O(log N) instead of O(N).
const NN_CELL_DEG = 0.005; // ≈ 500 m at this latitude
function buildNNGrid(oLat,oLon,N){
  const grid = new Map();
  for (let i=0;i<N;i++){
    const cx = Math.floor(oLon[i]/NN_CELL_DEG);
    const cy = Math.floor(oLat[i]/NN_CELL_DEG);
    const key = cx*100000+cy;
    let arr = grid.get(key);
    if (!arr){ arr=[]; grid.set(key,arr); }
    arr.push(i);
  }
  return grid;
}
function nnGridFind(grid, cLat, cLon, done, oLat, oLon){
  const ccx = Math.floor(cLon/NN_CELL_DEG);
  const ccy = Math.floor(cLat/NN_CELL_DEG);
  let bestI=-1, bestSq=Infinity;
  // Expand outward in concentric rings. Stop when the closest possible point
  // in the next ring (distance = ring * cell) is farther than our current best.
  for (let ring=0; ring<300; ring++){
    const ringMin = ring*NN_CELL_DEG;
    if (bestI!==-1 && ringMin*ringMin > bestSq) break;
    if (ring===0){
      const key = ccx*100000+ccy;
      const cell = grid.get(key);
      if (cell){
        for (let k=0;k<cell.length;k++){
          const i = cell[k];
          if (done[i]) continue;
          const dx=oLat[i]-cLat, dy=oLon[i]-cLon;
          const sq=dx*dx+dy*dy;
          if (sq<bestSq){bestSq=sq; bestI=i;}
        }
      }
    } else {
      // Iterate just the OUTER ring at radius `ring`
      for (let dx=-ring; dx<=ring; dx++){
        for (let dy=-ring; dy<=ring; dy++){
          if (dx!==-ring && dx!==ring && dy!==-ring && dy!==ring) continue;
          const key = (ccx+dx)*100000+(ccy+dy);
          const cell = grid.get(key);
          if (!cell) continue;
          for (let k=0;k<cell.length;k++){
            const i = cell[k];
            if (done[i]) continue;
            const ddx=oLat[i]-cLat, ddy=oLon[i]-cLon;
            const sq=ddx*ddx+ddy*ddy;
            if (sq<bestSq){bestSq=sq; bestI=i;}
          }
        }
      }
    }
  }
  return bestI;
}

// --- Step 4: Restoration plan ---
// Each crew starts at a depot, picks nearest unrepaired outage, repairs (fixed time + travel),
// moves to next. Total restoration time = max(crew time across all crews).
function planRestoration(){
  if (!storm){status.textContent="Simulate a storm first.";return;}
  const seed=+seedI.value|0;
  const rnd=mulberry32(seed*31+99);
  const M=+cS.value;
  const N=storm.outages.length;
  // --- Realism parameters ---
  // When realistic mode is on, the scheduler models the real-world delays that
  // turn a "12-hour theoretical" restoration into the 2–7-day reality reported
  // in Eversource storm filings: assessment delay, night-ops downtime, tiered
  // priority, longer per-repair time.
  const realistic = document.getElementById('realisticMode').checked;
  const REPAIR_HRS       = realistic ? 3.0  : 1.5;   // diagnosis + repair + verify
  const TRAVEL_MPH       = realistic ? 25   : 30;    // slower under storm conditions
  const ASSESSMENT_DELAY = realistic ? 12   : 0;     // hours before any crew dispatches
  const WORKDAY_HOURS    = realistic ? 14   : 24;    // 6 am – 8 pm daylight ops
  function clampToWorkday(t){
    if (!realistic) return t;
    const dayN = Math.floor(t/24);
    const inDay = t - dayN*24;
    return inDay > WORKDAY_HOURS ? (dayN+1)*24 : t;
  }
  status.textContent=`Planning restoration: ${M.toLocaleString()} crews, ${N.toLocaleString()} jobs${realistic?' (realistic mode)':''}…`;

  // Depot placement
  //  - small M: k-means on outage points (gives nicely spread depots)
  //  - large M: skip k-means (cost grows as O(M*N) per iter); seed depots from outage points cycling
  let depots;
  if (M <= 200) {
    const depotPts=storm.outages.map(o=>({lat:o.lat,lon:o.lon,w:1}));
    depots=kmeans(depotPts,Math.min(M,depotPts.length),countyGeo,rnd,12);
  } else {
    depots=[];
    for (let i=0;i<M;i++){const src=storm.outages[i%N]; depots.push({lat:src.lat,lon:src.lon});}
  }
  const crews=depots.map((d,i)=>({depot:d,color:PALETTE[i%PALETTE.length],jobs:[],time:ASSESSMENT_DELAY,curLat:d.lat,curLon:d.lon}));

  // Fast scheduler — grid-hash O(log N) nearest-outage lookup, min-heap on crews.
  const oLat=new Float64Array(N), oLon=new Float64Array(N);
  for (let i=0;i<N;i++){oLat[i]=storm.outages[i].lat; oLon[i]=storm.outages[i].lon;}
  const done=new Uint8Array(N);

  // Realistic mode: schedule in tiers. Tier 1 = critical facilities; the rest
  // is processed only after every critical site has a crew assigned.
  let criticalCount = 0;
  if (realistic) {
    for (let i=0;i<N;i++){
      if (storm.outages[i].critical) criticalCount++;
      else done[i]=1;   // hide non-critical for the first tier
    }
  }
  let remaining = realistic ? criticalCount : N;
  let inCriticalPhase = realistic && criticalCount > 0;

  function hPush(h,item){h.push(item);let i=h.length-1;while (i>0){const p=(i-1)>>1;if (h[p][0]<=h[i][0]) break;const t=h[i];h[i]=h[p];h[p]=t;i=p;}}
  function hPop(h){const top=h[0];const last=h.pop();if (h.length>0){h[0]=last;let i=0;const L=h.length;while (true){const l=2*i+1,r=2*i+2;let s=i;if (l<L&&h[l][0]<h[s][0]) s=l;if (r<L&&h[r][0]<h[s][0]) s=r;if (s===i) break;const t=h[i];h[i]=h[s];h[s]=t;i=s;}}return top;}

  const nnGrid=buildNNGrid(oLat,oLon,N);

  const heap=[];
  for (let c=0;c<crews.length;c++) hPush(heap,[ASSESSMENT_DELAY,c]);

  while (remaining>0){
    const top=hPop(heap);
    const ci=top[1];
    const crew=crews[ci];
    const cLat=crew.curLat, cLon=crew.curLon;
    const bestI = nnGridFind(nnGrid, cLat, cLon, done, oLat, oLon);
    if (bestI===-1){
      // Either critical tier just drained, or nothing else to assign.
      if (inCriticalPhase) {
        // Open up the non-critical outages and keep going.
        for (let i=0;i<N;i++) if (!storm.outages[i].critical) done[i]=0;
        remaining = N - criticalCount;
        inCriticalPhase = false;
        hPush(heap,[crew.time,ci]); // crew goes back into the queue at its current time
        continue;
      }
      break;
    }
    done[bestI]=1; remaining--;
    if (remaining===0 && inCriticalPhase){
      for (let i=0;i<N;i++) if (!storm.outages[i].critical) done[i]=0;
      remaining = N - criticalCount;
      inCriticalPhase = false;
    }
    const o=storm.outages[bestI];
    const d=dist({lat:cLat,lon:cLon},o);
    let eta = crew.time + d/TRAVEL_MPH + REPAIR_HRS;
    eta = clampToWorkday(eta);   // push past overnight if it'd extend into the night band
    crew.jobs.push({outageIdx:bestI,o,etaFinish:eta});
    crew.time=eta; crew.curLat=o.lat; crew.curLon=o.lon;
    hPush(heap,[eta,ci]);
  }

  let totalTime=0;
  for (const c of crews) if (c.time>totalTime) totalTime=c.time;
  plan={crews,totalTime};

  // Render — visual style preserved at every scale, with graceful degradation
  // to Canvas-rendered shapes once DOM markers would dominate the frame budget.
  //  - Depots: SVG divIcons up to 500 crews; Canvas squares above.
  //  - Numbered repair circles: SVG divIcons until 1,500 total are placed;
  //    after that, Canvas dots (still colored per crew, just unnumbered).
  depotLayer.clearLayers(); repairLayer.clearLayers();

  const DEPOT_SVG_LIMIT = 500;
  if (crews.length <= DEPOT_SVG_LIMIT) {
    crews.forEach(c=>{
      const html=`<div style="width:14px;height:14px;background:${c.color};border:2px solid #111;"></div>`;
      L.marker([c.depot.lat,c.depot.lon],{icon:L.divIcon({className:'crew-depot',html,iconSize:[14,14],iconAnchor:[7,7]})}).addTo(depotLayer)
        .bindTooltip(`Crew ${c.color} depot`);
    });
  } else {
    // Canvas square via a circleMarker (Leaflet has no canvas square primitive)
    crews.forEach(c=>{
      L.circleMarker([c.depot.lat,c.depot.lon],{
        renderer:repairCanvas, radius:5, color:"#111", weight:1.2, fillColor:c.color, fillOpacity:1, interactive:false
      }).addTo(depotLayer);
    });
  }

  const NUMBERED_BUDGET = 1500;
  let numbered = 0;
  for (const c of crews){
    for (let order=0; order<c.jobs.length; order++){
      const j = c.jobs[order];
      if (order < 30 && numbered < NUMBERED_BUDGET) {
        const html = `<div style="width:14px;height:14px;border-radius:50%;background:${c.color};color:white;font-size:9px;font-weight:bold;line-height:14px;text-align:center;border:1px solid #111">${order+1}</div>`;
        L.marker([j.o.lat,j.o.lon],{icon:L.divIcon({className:'',html,iconSize:[14,14],iconAnchor:[7,7]}),interactive:false}).addTo(repairLayer);
        numbered++;
      } else {
        L.circleMarker([j.o.lat,j.o.lon],{
          renderer:repairCanvas,radius:3,color:"#111",weight:0.5,fillColor:c.color,fillOpacity:0.95,interactive:false
        }).addTo(repairLayer);
      }
    }
  }

  // Display: hours under a day, days + hours past 24 h (realistic scenarios run for days).
  function fmtDuration(hrs){
    if (hrs < 24) return hrs.toFixed(1) + " h";
    const d = Math.floor(hrs/24);
    const h = Math.round(hrs - d*24);
    return d + " d " + h + " h";
  }
  restoreTimeEl.textContent = fmtDuration(totalTime);
  const modeNote = realistic ? ' · realistic mode' : ' · optimistic baseline';
  restoreDetailEl.textContent=`${N.toLocaleString()} jobs across ${crews.length.toLocaleString()} crews${modeNote}`;
  status.textContent=`Restoration: ${crews.length.toLocaleString()} crews finish all ${N.toLocaleString()} repairs in ${fmtDuration(totalTime)}${realistic?' (includes 12 h assessment + overnight downtime)':''}.`;
}

  affectedEl.textContent='0'; affectedPctEl.textContent='0% of county population';
  outageCountEl.textContent='0'; restoreTimeEl.textContent='—'; restoreDetailEl.textContent='— customers restored';
  status.textContent='Storm cleared.';

// --- Export: GeoJSON + Shapefile ---
// Convert the in-memory simulation state to standard GIS layers.
function buildFeatureCollections(){
  const seed = +seedI.value|0;
  const stamp = new Date().toISOString().replace(/[:.]/g,'-');

  // Layer 1: substations (Point) — sub_id, lat, lon, pop_serv, n_feeders, color_hex
  const subsFC = { type:"FeatureCollection", features: substations.map((s,i)=>({
    type:"Feature",
    geometry:{ type:"Point", coordinates:[+s.lon.toFixed(6), +s.lat.toFixed(6)] },
    properties:{
      sub_id: i+1,
      lat: +s.lat.toFixed(6),
      lon: +s.lon.toFixed(6),
      pop_serv: Math.round(s.popServed),
      n_feeders: feeders.filter(f=>f.subIdx===i).length,
      color_hex: s.color,
    },
  })) };

  // Layer 2: feeders (LineString)
  const feedsFC = { type:"FeatureCollection", features: feeders.map((f,i)=>{
    let lengthKm = 0;
    for (let k=1;k<f.pts.length;k++){
      const a={lat:f.pts[k-1][0],lon:f.pts[k-1][1]}, b={lat:f.pts[k][0],lon:f.pts[k][1]};
      lengthKm += dist(a,b)*1.60934;
    }
    return {
      type:"Feature",
      geometry:{ type:"LineString", coordinates: f.pts.map(p=>[+p[1].toFixed(6),+p[0].toFixed(6)]) },
      properties:{
        feed_id: i+1,
        sub_id: f.subIdx+1,
        pop_serv: Math.round(f.popServed),
        length_km: +lengthKm.toFixed(3),
        color_hex: f.color,
      },
    };
  }) };

  // Layer 3: laterals (LineString)
  const latsFC = { type:"FeatureCollection", features: laterals.map((l,i)=>{
    let lengthKm = 0;
    for (let k=1;k<l.pts.length;k++){
      const a={lat:l.pts[k-1][0],lon:l.pts[k-1][1]}, b={lat:l.pts[k][0],lon:l.pts[k][1]};
      lengthKm += dist(a,b)*1.60934;
    }
    return {
      type:"Feature",
      geometry:{ type:"LineString", coordinates: l.pts.map(p=>[+p[1].toFixed(6),+p[0].toFixed(6)]) },
      properties:{
        lat_id: i+1,
        feed_id: l.feederIdx+1,
        pop_serv: Math.round(l.popServed),
        length_km: +lengthKm.toFixed(3),
      },
    };
  }) };

  // Layer 4: outages (Point) — only if a storm has been simulated
  const outFC = { type:"FeatureCollection", features: (storm? storm.outages: []).map((o,i)=>({
    type:"Feature",
    geometry:{ type:"Point", coordinates:[+o.lon.toFixed(6), +o.lat.toFixed(6)] },
    properties:{
      out_id: i+1,
      seed,
      kind: o.kind === 'f' ? 'feeder' : 'lateral',
      feed_id: o.kind==='f' ? o.fi+1 : -1,
      lat_id:  o.kind==='l' ? o.li+1 : -1,
      pop_loss: Math.round(o.popLoss),
    },
  })) };

  // Layer 5: restoration plan (Point) — only if a plan has been computed
  const planFC = { type:"FeatureCollection", features: [] };
  if (plan) {
    plan.crews.forEach((c,ci)=>{
      c.jobs.forEach((j,order)=>{
        planFC.features.push({
          type:"Feature",
          geometry:{ type:"Point", coordinates:[+j.o.lon.toFixed(6), +j.o.lat.toFixed(6)] },
          properties:{
            job_ord: order+1,
            crew_id: ci+1,
            out_id:  j.outageIdx+1,
            eta_h: +j.etaFinish.toFixed(3),
            depot_lat: +c.depot.lat.toFixed(6),
            depot_lon: +c.depot.lon.toFixed(6),
          },
        });
      });
    });
  }

  const manifest = {
    schema_version: 1,
    generator: "hartford_county_simulation",
    generated_utc: new Date().toISOString(),
    crs: "EPSG:4326",
    inputs: {
      seed,
      substations: +kS.value,
      feeders_per_sub: +fS.value,
      storm_outages: +oS.value,
      repair_crews: +cS.value,
    },
    counts: {
      substations: substations.length,
      feeders: feeders.length,
      laterals: laterals.length,
      outages: outFC.features.length,
      restoration_jobs: planFC.features.length,
    },
  };

  return { subsFC, feedsFC, latsFC, outFC, planFC, manifest, stamp };
}

function downloadBlob(blob, filename){
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url; a.download = filename; document.body.appendChild(a); a.click();
  setTimeout(()=>{ document.body.removeChild(a); URL.revokeObjectURL(url); }, 0);
}

async function exportShapefile(){
  if (!substations.length){ status.textContent="Generate the grid first."; return; }
  if (typeof shpwrite === 'undefined' || typeof JSZip === 'undefined'){
    status.textContent = "Shapefile libraries did not load. Use Download GeoJSON instead.";
    return;
  }
  const { subsFC, feedsFC, latsFC, outFC, planFC, manifest, stamp } = buildFeatureCollections();
  const layers = [
    { name:"substations",     fc: subsFC,  hasGeom: subsFC.features.length>0 },
    { name:"feeders",         fc: feedsFC, hasGeom: feedsFC.features.length>0 },
    { name:"laterals",        fc: latsFC,  hasGeom: latsFC.features.length>0 },
    { name:"outages",         fc: outFC,   hasGeom: outFC.features.length>0 },
    { name:"restoration_plan",fc: planFC,  hasGeom: planFC.features.length>0 },
  ];
  const master = new JSZip();
  master.file("manifest.json", JSON.stringify(manifest, null, 2));
  for (const layer of layers) {
    if (!layer.hasGeom) continue;
    try {
      const blob = shpwrite.zip(layer.fc); // returns base64 string in 0.3.x
      // shp-write 0.3.x returns a base64-encoded zip. Decode and unpack into master zip.
      const raw = atob(blob);
      const inner = await JSZip.loadAsync(raw, { binary:true });
      for (const path in inner.files) {
        const f = inner.files[path];
        if (!f.dir) {
          const data = await f.async("uint8array");
          // Prefix with layer name to avoid collisions
          master.file(`${layer.name}/${path}`, data);
        }
      }
    } catch(e) {
      console.warn(`shp-write failed on ${layer.name}:`, e);
    }
  }
  const blob = await master.generateAsync({ type:"blob" });
  downloadBlob(blob, `hartford_grid_seed${manifest.inputs.seed}_${stamp}_shp.zip`);
  status.textContent = `Exported shapefile bundle (${manifest.counts.substations} subs, ${manifest.counts.feeders} feeders, ${manifest.counts.laterals} laterals).`;
}

document.getElementById('exportShapefile').onclick = exportShapefile;

// --- Recommendation: smallest crew count that's within tolerance of the time floor ---
// Pure scheduler — same algorithm as planRestoration() but no DOM rendering,
// no PALETTE color tagging, no plan-state mutation. Returns just totalTime.
function scheduleOnly(M, outages){
  const N = outages.length;
  if (N === 0 || M === 0) return 0;
  // Mirror the realism settings used by planRestoration so the recommendation
  // matches the actual restoration time the user will see.
  const realistic = document.getElementById('realisticMode').checked;
  const REPAIR_HRS       = realistic ? 3.0  : 1.5;
  const TRAVEL_MPH       = realistic ? 25   : 30;
  const ASSESSMENT_DELAY = realistic ? 12   : 0;
  const WORKDAY_HOURS    = realistic ? 14   : 24;
  function clampToWorkday(t){
    if (!realistic) return t;
    const dayN = Math.floor(t/24);
    const inDay = t - dayN*24;
    return inDay > WORKDAY_HOURS ? (dayN+1)*24 : t;
  }
  // Depots: small M → distinct k-means clusters; large M → cycle outage points.
  let depots;
  if (M <= 200) {
    const rnd = mulberry32((+seedI.value|0)*31 + 99);
    const pts = outages.map(o=>({lat:o.lat,lon:o.lon,w:1}));
    depots = kmeans(pts, Math.min(M, pts.length), countyGeo, rnd, 8);
  } else {
    depots = [];
    for (let i=0;i<M;i++){const s=outages[i%N]; depots.push({lat:s.lat,lon:s.lon});}
  }
  const crews = depots.map(d=>({time:ASSESSMENT_DELAY, curLat:d.lat, curLon:d.lon}));
  const oLat = new Float64Array(N), oLon = new Float64Array(N);
  for (let i=0;i<N;i++){oLat[i]=outages[i].lat; oLon[i]=outages[i].lon;}
  const done = new Uint8Array(N);
  // Tier 1 (critical-facility) first if realistic.
  let criticalCount = 0;
  if (realistic) {
    for (let i=0;i<N;i++){
      if (outages[i].critical) criticalCount++;
      else done[i] = 1;
    }
  }
  let remaining = realistic ? criticalCount : N;
  let inCriticalPhase = realistic && criticalCount > 0;
  function hPush(h,it){h.push(it);let i=h.length-1;while (i>0){const p=(i-1)>>1;if (h[p][0]<=h[i][0]) break;const t=h[i];h[i]=h[p];h[p]=t;i=p;}}
  function hPop(h){const top=h[0];const last=h.pop();if (h.length>0){h[0]=last;let i=0;const L=h.length;while (true){const l=2*i+1,r=2*i+2;let s=i;if (l<L&&h[l][0]<h[s][0]) s=l;if (r<L&&h[r][0]<h[s][0]) s=r;if (s===i) break;const t=h[i];h[i]=h[s];h[s]=t;i=s;}}return top;}
  const nnGrid=buildNNGrid(oLat,oLon,N);
  const heap=[];
  for (let c=0;c<crews.length;c++) hPush(heap,[ASSESSMENT_DELAY,c]);
  while (remaining>0){
    const [_,ci]=hPop(heap); const crew=crews[ci];
    const cLat=crew.curLat, cLon=crew.curLon;
    const bestI = nnGridFind(nnGrid, cLat, cLon, done, oLat, oLon);
    if (bestI===-1){
      if (inCriticalPhase){
        for (let i=0;i<N;i++) if (!outages[i].critical) done[i]=0;
        remaining = N - criticalCount;
        inCriticalPhase = false;
        hPush(heap,[crew.time,ci]);
        continue;
      }
      break;
    }
    done[bestI]=1; remaining--;
    if (remaining===0 && inCriticalPhase){
      for (let i=0;i<N;i++) if (!outages[i].critical) done[i]=0;
      remaining = N - criticalCount;
      inCriticalPhase = false;
    }
    const o=outages[bestI];
    const d=dist({lat:cLat,lon:cLon}, o);
    let eta = crew.time + d/TRAVEL_MPH + REPAIR_HRS;
    eta = clampToWorkday(eta);
    crew.time=eta; crew.curLat=o.lat; crew.curLon=o.lon;
    hPush(heap,[eta,ci]);
  }
  let t=0; for (const c of crews) if (c.time>t) t=c.time;
  return t;
}

function recommendCrewCount(){
  if (!storm || !storm.outages.length){
    status.textContent="Simulate a storm first.";
    document.getElementById('recCrewDetail').textContent='— simulate a storm first';
    return;
  }
  const N = storm.outages.length;
  status.textContent = `Searching for optimal crew count over ${N.toLocaleString()} jobs…`;
  // Floor time: M=N is the practical minimum (one crew per outage, no queueing).
  // Beyond M=N extra crews idle, so we never need to test higher.
  const floorTime = scheduleOnly(N, storm.outages);
  const tolerance = 1.15;        // accept restoration times up to 15% above floor
  const target = floorTime * tolerance;
  // Binary search for smallest M with scheduleOnly(M) <= target.
  // Cache to avoid recomputing.
  const cache = new Map();
  cache.set(N, floorTime);
  function tAt(M){
    if (cache.has(M)) return cache.get(M);
    const t = scheduleOnly(M, storm.outages);
    cache.set(M, t);
    return t;
  }
  let lo=1, hi=N;
  while (lo<hi){
    const mid = (lo+hi)>>1;
    if (tAt(mid) <= target) hi = mid;
    else lo = mid+1;
  }
  const recM = lo;
  const recT = tAt(recM);

  function fmtH(hrs){
    if (hrs < 24) return hrs.toFixed(1) + " h";
    const d = Math.floor(hrs/24);
    const h = Math.round(hrs - d*24);
    return d + " d " + h + " h";
  }
  document.getElementById('recCrewCount').textContent = recM.toLocaleString();
  document.getElementById('recCrewDetail').innerHTML =
    `${fmtH(recT)} restoration · within 15% of ${fmtH(floorTime)} floor`;
  const applyBtn = document.getElementById('applyRecCrews');
  applyBtn.style.display = 'block';
  applyBtn.onclick = () => {
    cS.value = Math.min(recM, +cS.max);
    cV.textContent = cS.value;
    planRestoration();
  };
  status.textContent = `Recommended ${recM.toLocaleString()} crews → ${fmtH(recT)} (floor ${fmtH(floorTime)} with ${N.toLocaleString()} crews).`;
}

document.getElementById('recommendCrews').onclick = recommendCrewCount;
